```javascript
const structuredData = {

  "startDate": "2026-06-07T09:00:00-04:00",
  "eventStatus": "https://schema.org/EventScheduled",

      "@type": "PostalAddress",
      "addressLocality": "Marquette",
      "addressRegion": "MI",
      "addressCountry": "US"
    }
  },
  "offers": {
    "@type": "Offer",
    "price": "0",
    "priceCurrency": "USD",
    "availability": "https://schema.org/InStock",
    "url": Astro.site?.toString() ?? "https://mkultragravel.com"
  },
  "organizer": {
    "@type": "Organization",
    "name": "MK Ultra Gravel",
    "url": Astro.site?.toString() ?? "https://mkultragravel.com"
  },
  "description": "100 miles of rowdy, technical gravel through Michigan's Upper Peninsula. Free ride. Mass start. No mercy.",
  "url": Astro.site?.toString() ?? "https://mkultragravel.com"

```

**In the `<head>` section**, add the script tag AFTER the Twitter Card meta tags block and BEFORE the `<link rel="icon">` line:

```html
<!-- Structured Data -->
<script type="application/ld+json" set:html={JSON.stringify(structuredData)} />
```

IMPORTANT:
- Use `set:html={JSON.stringify(structuredData)}` -- this is the correct Astro pattern for injecting JSON into a script tag. Do NOT use `define:vars`. Do NOT use template literals inside the tag.
- Use `Astro.site?.toString()` for URLs (with nullish coalescing fallback) to derive URLs dynamically from the site config.
- Place the `<script>` in `<head>`, NOT in `<body>`.
- The `<script>` tag uses a self-closing `/>` format with `set:html` -- Astro handles this correctly.
- Do NOT change any existing tags or elements in the file.
- Do NOT install any library.
