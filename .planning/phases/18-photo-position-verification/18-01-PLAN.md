4. If verification fails (any mismatch found):
   - Run the FULL pipeline `node scripts/generate-data.js` (not just match-photos.js alone) to regenerate photos.json WITH width/height fields intact
   - Re-run the comparison to confirm mismatches are resolved
   - Stage the updated photos.json

IMPORTANT: Do NOT commit a photos.json that is missing width/height fields. The match-photos.js script alone outputs {filename, lat, lon, mi, source}. The generate-thumbnails.js step (run by generate-data.js) enriches it with {width, height}. Always restore or use the full pipeline.

Run this inline Node.js check and confirm output shows 0 mismatches:
```
node -e "
const photos = require('./public/data/photos.json');
const {photoManifest} = require('./scripts/photo-manifest.js');
let issues = 0;
if (photos.length !== photoManifest.length) { console.log('COUNT MISMATCH:', photos.length, 'vs', photoManifest.length); issues++; }

  const p = photos.find(x => x.filename === m.filename);
  if (!p) { console.log('MISSING:', m.filename); issues++; continue; }
  if (p.mi !== m.mi) { console.log('MI MISMATCH:', m.filename, 'expected:', m.mi, 'got:', p.mi); issues++; }
}
console.log(issues === 0 ? 'PASS: All 54 entries verified' : 'FAIL: ' + issues + ' issue(s)');
"
```
