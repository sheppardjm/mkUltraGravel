   ```
     width: photo.width,
     height: photo.height,
     msrc: `/images/thumbs/${photo.filename.replace(/\.(jpg|jpeg|png|avif)$/i, '.webp')}`,
     alt: `Route photo at mile ${photo.mi}`,
   }));
   ```

5. AFTER the dataSource array, ADD lightbox initialization:
   ```
   const lightbox = new PhotoSwipeLightbox({
     dataSource,
     pswpModule: () => import('photoswipe'),
     bgOpacity: 0.95,
     showHideAnimationType: 'fade',
   });
   lightbox.init();
   ```

6. REPLACE the `photoMarkers` array creation with per-photo icon + click handler:
   ```
   const photoMarkers = photos.map((photo: { filename: string; lat: number; lon: number; mi: number }, index: number) => {
     const icon = L.divIcon({
       className: 'photo-marker',
       html: `<img
         src="/images/thumbs/${photo.filename.replace(/\.(jpg|jpeg|png|avif)$/i, '.webp')}"
         alt="Route photo at mile ${photo.mi}"
         style="width:48px;height:48px;object-fit:cover;border:2px solid #22d3ee;border-radius:3px;display:block;"
         loading="lazy"
       />`,
       iconSize: [52, 52],
       iconAnchor: [26, 26],
     });

     const marker = L.marker([photo.lat, photo.lon], { icon });
     marker.on('click', () => {
       lightbox.loadAndOpen(index);
     });
     return marker;
   });
   ```
   Note: NO `.bindPopup()` call. Clicking opens lightbox directly.
