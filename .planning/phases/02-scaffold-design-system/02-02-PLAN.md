  1. Update `src/styles/global.css` — replace the empty `@theme {}` block with the full
     design token system. Keep the existing @layer declaration and @import lines unchanged.

     ```css
     @theme {
       /* Colors — dark brutalist palette */
       --color-bg-base: oklch(0.10 0.01 250);
       --color-bg-surface: oklch(0.14 0.01 250);
       --color-bg-elevated: oklch(0.18 0.01 250);
       --color-accent-green: oklch(0.85 0.24 145);
       --color-accent-red: oklch(0.45 0.22 25);
       --color-accent-white: oklch(0.92 0.01 90);
       --color-text-body: oklch(0.85 0.01 90);
       --color-text-muted: oklch(0.55 0.01 90);
       --color-border: oklch(0.25 0.01 250);

       /* Fonts — set by Astro Fonts API via CSS variables */
       --font-sans: var(--font-mono);
       --font-mono: "Space Mono", ui-monospace, monospace;
       --font-display: "Special Elite", serif;
     }
     ```

     Add a `@layer base` block AFTER the @theme block:

     ```css
     @layer base {
       html {
         background-color: var(--color-bg-base);
         color: var(--color-text-body);
         font-family: var(--font-mono);
         -webkit-font-smoothing: antialiased;
         -moz-osx-font-smoothing: grayscale;
       }

       body {
         min-height: 100vh;
         line-height: 1.6;
       }

       h1, h2, h3, h4, h5, h6 {
         font-family: var(--font-display);
         color: var(--color-accent-white);
         line-height: 1.1;
         letter-spacing: 0.02em;
       }

       a {
         color: var(--color-accent-green);
         text-decoration-thickness: 1px;
         text-underline-offset: 2px;
       }

       ::selection {
         background-color: var(--color-accent-green);
         color: var(--color-bg-base);
       }
     }
     ```

  2. Update `astro.config.mjs` — add the Fonts API configuration:

     ```javascript
     import { defineConfig, fontProviders } from "astro/config";
     import tailwindcss from "@tailwindcss/vite";

     export default defineConfig({
       vite: {
         plugins: [tailwindcss()],
       },
       fonts: [
         {
           name: "Space Mono",
           cssVariable: "--font-mono",
           provider: fontProviders.google(),
           weights: [400, 700],
           styles: ["normal", "italic"],
           subsets: ["latin"],
         },
         {
           name: "Special Elite",
           cssVariable: "--font-display",
           provider: fontProviders.google(),
           weights: [400],
           styles: ["normal"],
           subsets: ["latin"],
         },
       ],
     });
     ```

     IMPORTANT: The Astro Fonts API is built into Astro 6 — no separate package needed.
     The `fontProviders` import comes from "astro/config". If this import fails at dev time,
     fall back to manual Google Fonts `<link>` preloads in BaseLayout.astro head:
     ```html
     <link rel="preconnect" href="https://fonts.googleapis.com" />
     <link rel="preconnect" href="https://fonts.gstatic.com" crossorigin />
     <link href="https://fonts.googleapis.com/css2?family=Space+Mono:ital,wght@0,400;0,700;1,400&family=Special+Elite&display=swap" rel="stylesheet" />
     ```
     Only use this fallback if the Fonts API does not work.
