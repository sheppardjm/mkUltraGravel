```typescript
annotations.sectors.forEach((sector: { name: string; stars: number; lengthMi: number; track: { lat: number; lon: number }[] }, i: number) => {
  const sectorLatlngs = sector.track.map((pt) => [pt.lat, pt.lon] as [number, number]);
  const originalStyle = {
    color: starColors[sector.stars] || '#ffffff',
    weight: 5,
    opacity: 0.9
  };

  const poly = L.polyline(sectorLatlngs, { ...originalStyle })
    .bindPopup(
      `<strong>${sector.name}</strong><br>` +
      `${'★'.repeat(sector.stars)}${'☆'.repeat(5 - sector.stars)}<br>` +
      `${sector.lengthMi.toFixed(1)} mi`,
      { className: 'dark-popup' }
    )
    .addTo(map);

  poly.on('mouseover', () => {
    window.dispatchEvent(new CustomEvent('map:sectorHover', { detail: { sectorIndex: i } }));
    poly.setStyle({ weight: 7, opacity: 1.0 });
  });
  poly.on('mouseout', () => {
    window.dispatchEvent(new CustomEvent('map:sectorHover', { detail: { sectorIndex: null } }));
    poly.setStyle(originalStyle);
  });
  poly.on('click', () => {
    window.dispatchEvent(new CustomEvent('map:sectorClick', { detail: { sectorIndex: i } }));
  });

  sectorPolylines.push(poly);

  // Persistent star-rating badge at sector midpoint
  const mid = Math.floor(sector.track.length / 2);
  const midPt = sector.track[mid];
  const starStr = '★'.repeat(sector.stars) + '☆'.repeat(5 - sector.stars);
  const badge = L.marker([midPt.lat, midPt.lon], {
    icon: L.divIcon({
      className: 'sector-badge',
      html: `<span style="color:${starColors[sector.stars] || '#ffffff'}">${starStr}</span>`,
      iconSize: [null, null] as any,
      iconAnchor: [0, 0]
    }),
    interactive: false
  });
  sectorBadges.addLayer(badge);
});
```

3. **Add `elevation:sectorClick` listener** AFTER the crosshair event listeners (from 13-01), using the same `signal` for cleanup:

```typescript
window.addEventListener('elevation:sectorClick', ((e: Event) => {
  const { sectorIndex } = (e as CustomEvent<{sectorIndex: number}>).detail;
  const poly = sectorPolylines[sectorIndex];
  if (poly) {
    // Dim all other sectors, highlight the clicked one
    sectorPolylines.forEach((p: any, i: number) => {
      if (i === sectorIndex) {
        p.setStyle({ weight: 7, opacity: 1.0 });
      } else {
        p.setStyle({ weight: 5, opacity: 0.4 });
      }
    });
    // Zoom to the sector bounds
    map.flyToBounds(poly.getBounds(), { maxZoom: 14, padding: [40, 40] });
    // Restore all polylines after flyToBounds animation ends (~250ms default)
    map.once('moveend', () => {
      sectorPolylines.forEach((p: any, i: number) => {
        const sector = annotations.sectors[i];
        p.setStyle({
          color: starColors[sector.stars] || '#ffffff',
          weight: 5,
          opacity: 0.9
        });
      });
    });
  }
}) as EventListener, { signal });
```
